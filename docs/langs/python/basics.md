```python
# String
x = "Hello World"
x = str("Hello World")

# Number
x = 20
x = int(20)
x = 20.5
x = float(20.5)
x = 1j
x = complex(1j)

# List
x = ["apple", "banana", "cherry"]
x = list(("apple", "banana", "cherry"))

# Tuple
x = ("apple", "banana", "cherry")
x = tuple(("apple", "banana", "cherry"))

# Range
x = range(6)

# Dictionary
x = {"name" : "John", "age" : 36}
x = dict(name="John", age=36)

# Set
x = {"apple", "banana", "cherry"}
x = set(("apple", "banana", "cherry"))

# Frozen Set
x = frozenset({"apple", "banana", "cherry"})
x = frozenset(("apple", "banana", "cherry"))

# Boolean
x = True
x = bool(5)

#Bytes
x = b"Hello"
x = bytes(5)
x = bytearray(5)
x = memoryview(bytes(5))
```
